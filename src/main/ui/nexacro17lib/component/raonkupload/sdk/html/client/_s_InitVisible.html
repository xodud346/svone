K Upload :: InitVisible

InitVisible

업로드를 화면에 표시할지 여부를 설정합니다.

remarks

"true"로 설정 시 업로드를 화면에 표시하고, "false"로 설정 시 표시하지 않습니다.
기본값은 "true" 입니다.

sample code

<script type="text/javascript" src="raonkupload/js/raonkupload.js"></script>	

<!-- ..... 생략 ..... -->
 
<div style="width:800px;height:400px">  
    <script type="text/javascript">
        // upload1 업로드는 화면에 표시가 됩니다.
        var raonkParam1 = {
            Id: 'upload1',
            .....
            InitVisible: true
        }
        
        new RAONKUpload(raonkParam1);
    </script>

    <script type="text/javascript">
        // upload2 업로드는 화면에 표시되지 않습니다.
        var raonkParam2 = {
            Id: 'upload2',
            .....
            InitVisible: false
        }
        
        new RAONKUpload(raonkParam2);
    </script>
</div>
